#!/usr/bin/env python3 """Generate PWA icons for Scott eRide Strike app""" import base64, os # SVG icon - lightning bolt on dark background with Scott yellow svg_192 = '''''' svg_512 = svg_192.replace('width="192" height="192" viewBox="0 0 192 192"', 'width="512" height="512" viewBox="0 0 192 192"') os.makedirs('icons', exist_ok=True) with open('icons/icon-192.svg', 'w') as f: f.write(svg_192) with open('icons/icon-512.svg', 'w') as f: f.write(svg_512) # Try to convert with cairosvg or use Python Pillow fallback try: import cairosvg cairosvg.svg2png(bytestring=svg_192.encode(), write_to='icons/icon-192.png', output_width=192, output_height=192) cairosvg.svg2png(bytestring=svg_512.encode(), write_to='icons/icon-512.png', output_width=512, output_height=512) print("Icons generated with cairosvg") except ImportError: # Fallback: generate with Pillow try: from PIL import Image, ImageDraw for size in [192, 512]: img = Image.new('RGBA', (size, size), (0,0,0,0)) d = ImageDraw.Draw(img) # Rounded rect background d.rounded_rectangle([0,0,size-1,size-1], radius=int(size*0.17), fill=(10,10,10,255)) # Lightning bolt scaled s = size / 192 pts = [(112*s,20*s),(72*s,100*s),(100*s,100*s),(80*s,172*s),(132*s,88*s),(104*s,88*s),(128*s,20*s)] d.polygon(pts, fill=(232,255,0,255)) img.save(f'icons/icon-{size}.png') print("Icons generated with Pillow") except ImportError: # Last resort: copy a minimal 1x1 PNG import struct, zlib def make_png(size, color=(10,10,10)): w = h = size raw = b'' for _ in range(h): raw += b'\x00' + bytes(color) * w compressed = zlib.compress(raw) def chunk(t, d): c = struct.pack('>I', len(d)) + t + d return c + struct.pack('>I', zlib.crc32(t+d) & 0xffffffff) png = b'\x89PNG\r\n\x1a\n' png += chunk(b'IHDR', struct.pack('>IIBBBBB', w, h, 8, 2, 0, 0, 0)) png += chunk(b'IDAT', compressed) png += chunk(b'IEND', b'') return png for s in [192, 512]: with open(f'icons/icon-{s}.png', 'wb') as f: f.write(make_png(s)) print("Icons generated (minimal fallback)")
